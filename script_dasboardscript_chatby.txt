function findChatwootActiveEl() {
  // 1) padrão mais comum: link com aria-current
  var a = document.querySelector('aside a[aria-current="page"]');
  if (a) return a;

  // 2) classes de router (alguns temas)
  var router = document.querySelector('aside .router-link-exact-active, aside .router-link-active');
  if (router) return router;

  // 3) fallback: tentar encontrar algum item com aparência de "active" via classes mais comuns do sidebar
  // (quando o item é <div role="button">)
  var possible = document.querySelector('aside nav ul li > a.bg-n-alpha-2, aside nav ul li > div.bg-n-alpha-2');
  if (possible) return possible;

  // 4) último fallback: nada encontrado
  return null;
}

function ensureNativeGroupExpanded(groupEl) {
  if (!groupEl) return;
  try { groupEl.setAttribute('aria-expanded', 'true'); } catch (e) {}
  var li = groupEl.closest ? groupEl.closest('li') : null;
  if (!li) return;
  var sub = li.querySelector(':scope > ul, ul.sidebar-group-children, ul');
  if (!sub) return;
  sub.style.display = '';
  sub.style.maxHeight = '';
  sub.style.height = '';
  sub.style.overflow = '';
  sub.style.visibility = '';
  sub.classList.remove('hidden', 'invisible', 'h-0', 'max-h-0', 'overflow-hidden');
  sub.classList.add('block');
}

function bindNativeSidebarBridge() {
  if (nativeSidebarBridgeBound) return;
  nativeSidebarBridgeBound = true;

  document.addEventListener('click', function(e) {
    var target = e.target.closest('aside nav a, aside nav div[role="button"], aside nav button');
    if (!target) return;

    // ignora menus custom
    if (target.id && target.id.indexOf('cw-custom-menu-') === 0) return;

    // se houver estado custom preso, limpa antes da navegação nativa
    if (!panelOpen) {
      forceCleanupNativeState();
      return;
    }

    var isNativeGroupButton = target.matches('div[role="button"], button');
    var isExpandedGroup = isNativeGroupButton && target.getAttribute('aria-expanded') === 'true';

    // Para grupos nativos (expandidos ou não), fecha o custom e reexecuta o clique
    // no próprio grupo, evitando precisar de 2 cliques para expandir tudo.
    if (isNativeGroupButton) {
      e.preventDefault();
      e.stopPropagation();
      e.stopImmediatePropagation();

      closeCustomPanelForNativeNavigation();
      forceCleanupNativeState();
      var clickedNativeGroup = target;
      requestAnimationFrame(function() {
        if (!clickedNativeGroup || !clickedNativeGroup.isConnected) return;
        // reenviar o clique apenas uma vez e deixar o Chatwoot controlar expansão
        clickedNativeGroup.dispatchEvent(new MouseEvent('click', {
          bubbles: true,
          cancelable: true,
          view: window
        }));
      });
      return;
    }

    // para links nativos normais e grupos fechados:
    // deixa o clique seguir e fecha o custom logo depois
    setTimeout(function() {
      if (!panelOpen) {
        forceCleanupNativeState();
        return;
      }
      closeCustomPanelForNativeNavigation();
      forceCleanupNativeState();
    }, 0);
  }, true);
}

function setMenuActiveLikeChatwoot(customEl) {
  if (!customEl) return;

  // 1) Limpa marcações dos CUSTOMS antes de marcar o novo
  var customEls = document.querySelectorAll(
    '#cw-custom-menu-kanban,#cw-custom-menu-conexoes,#cw-custom-menu-chats,#cw-custom-menu-projetos,#cw-custom-menu-permissoes,#cw-custom-menu-chatbot,#cw-custom-menu-permissions'
  );

  customEls.forEach(function(el) {
    el.removeAttribute('aria-current');
    el.classList.remove('router-link-active', 'router-link-exact-active', 'cw-custom-active');

    // remove estilos inline dos customs
    el.style.backgroundColor = '';
    el.style.color = '';
    el.style.fontWeight = '';
    el.style.borderRadius = '';
    el.style.boxShadow = '';
    el.style.outline = '';

    var li = el.closest('li');
    if (li) li.classList.remove('cw-custom-active-li');
  });

  // 2) Encontra o item original "ativo"
  var activeEl = findChatwootActiveEl();

  // 3) Captura o original só uma vez e neutraliza o visual dele (para não ficar 2 ativos)
  if (!cwPrevActiveEl && activeEl && activeEl !== customEl && !isCustomMenuEl(activeEl)) {
    cwPrevActiveEl = activeEl;
    cwPrevAria = activeEl.getAttribute('aria-current');
    cwPrevClass = activeEl.className;

    // salva o inline style atual do original para restaurar depois
    cwPrevInlineStyle = activeEl.getAttribute('style');

    // pega o LI do item original ativo (muitos temas pintam o LI, não o A)
    cwPrevActiveLi = activeEl.closest ? activeEl.closest('li') : null;
    cwPrevLiInlineStyle = cwPrevActiveLi ? cwPrevActiveLi.getAttribute('style') : null;

    // tenta achar um elemento interno que esteja com fundo (bg-n-alpha-2/hover preso)
    cwPrevActiveBgEl = null;
    if (cwPrevActiveLi) {
      cwPrevActiveBgEl =
      cwPrevActiveLi.querySelector('a.rounded-lg, div[role="button"].rounded-lg') ||
      cwPrevActiveLi.querySelector('a[title], div[role="button"][title]') ||
      null;
    }
    cwPrevBgInlineStyle = cwPrevActiveBgEl ? cwPrevActiveBgEl.getAttribute('style') : null;

    // neutraliza o fundo do wrapper interno também (remove o cinza preso)
    if (cwPrevActiveBgEl) {
      cwPrevActiveBgEl.style.backgroundColor = 'transparent';
      cwPrevActiveBgEl.style.boxShadow = 'none';
    }

    // snapshot do estilo computado do original ativo (para aplicar no custom)
    try {
      var cs = window.getComputedStyle(activeEl);
      window.__cwActiveStyleSnapshot = {
        backgroundColor: cs.backgroundColor,
        color: cs.color,
        fontWeight: cs.fontWeight,
        borderRadius: cs.borderRadius,
        boxShadow: cs.boxShadow,
        outline: cs.outline
      };
    } catch (e) {
      window.__cwActiveStyleSnapshot = null;
    }

    // tenta desmarcar (pode voltar pelo Vue)
    activeEl.removeAttribute('aria-current');
    activeEl.classList.remove('router-link-active', 'router-link-exact-active');

    // ✅ override visual: garante que o original NÃO pareça ativo enquanto o custom está aberto
    activeEl.style.backgroundColor = 'transparent';
    activeEl.style.boxShadow = 'none';
    activeEl.style.fontWeight = '400';
    activeEl.style.color = 'inherit';

    // ✅ override visual no LI também (remove fundo/realce do original)
    if (cwPrevActiveLi) {
     cwPrevActiveLi.style.backgroundColor = 'transparent';
     cwPrevActiveLi.style.boxShadow = 'none';
     }
  }

  // 4) Marca o custom como ativo
  customEl.setAttribute('aria-current', 'page');
  customEl.classList.add('router-link-active', 'router-link-exact-active', 'cw-custom-active');

  // 5) Aplica o estilo visual do original ativo no custom
  try {
    var snap = window.__cwActiveStyleSnapshot;
    if (snap) {
      customEl.style.backgroundColor = snap.backgroundColor;
      customEl.style.color = snap.color;
      customEl.style.fontWeight = snap.fontWeight;
      customEl.style.borderRadius = snap.borderRadius;
      customEl.style.boxShadow = snap.boxShadow;
      customEl.style.outline = snap.outline;
    }
  } catch (e) {}

  // 6) Marca o LI também (alguns temas aplicam fundo no <li>)
  var customLi = customEl.closest('li');
  if (customLi) customLi.classList.add('cw-custom-active-li');
}

function restoreChatwootActive() {

  // remove marcação dos customs
  var customActives = document.querySelectorAll(
    '#cw-custom-menu-kanban[aria-current],#cw-custom-menu-conexoes[aria-current],#cw-custom-menu-chats[aria-current],#cw-custom-menu-projetos[aria-current],#cw-custom-menu-permissoes[aria-current],#cw-custom-menu-chatbot[aria-current],#cw-custom-menu-permissions[aria-current]'
  );

  customActives.forEach(function(el) {
    el.removeAttribute('aria-current');
    el.classList.remove('router-link-active', 'router-link-exact-active', 'cw-custom-active');

    // remove estilos inline aplicados no "ativo" custom
    el.style.backgroundColor = '';
    el.style.color = '';
    el.style.fontWeight = '';
    el.style.borderRadius = '';
    el.style.boxShadow = '';
    el.style.outline = '';

    // se marcou o LI pai como ativo, limpe também
    var li = el.closest && el.closest('li');
    if (li) li.classList.remove('cw-custom-active-li');
  });

  // restaura o original (desfaz override visual)
  if (cwPrevActiveEl) {
    // restaura style inline original
    if (cwPrevInlineStyle !== null && cwPrevInlineStyle !== undefined) {
      cwPrevActiveEl.setAttribute('style', cwPrevInlineStyle);
    } else {
      cwPrevActiveEl.removeAttribute('style');
    }

    // restaura aria-current e className originais
    if (cwPrevAria) cwPrevActiveEl.setAttribute('aria-current', cwPrevAria);
    if (cwPrevClass) cwPrevActiveEl.className = cwPrevClass;
  }

  // restaura style inline do LI original
  if (cwPrevActiveLi) {
    if (cwPrevLiInlineStyle !== null && cwPrevLiInlineStyle !== undefined) {
      cwPrevActiveLi.setAttribute('style', cwPrevLiInlineStyle);
    } else {
    cwPrevActiveLi.removeAttribute('style');
    }
  }

  // restaura style do wrapper interno (se existia)
  if (cwPrevActiveBgEl) {
  if (cwPrevBgInlineStyle !== null && cwPrevBgInlineStyle !== undefined) {
    cwPrevActiveBgEl.setAttribute('style', cwPrevBgInlineStyle);
  } else {
    cwPrevActiveBgEl.removeAttribute('style');
  }
  }

  // limpa estado salvo
  cwPrevActiveEl = null;
  cwPrevAria = null;
  cwPrevClass = null;
  cwPrevInlineStyle = null;
  cwPrevActiveLi = null;
  cwPrevLiInlineStyle = null;
  cwPrevActiveBgEl = null;
  cwPrevBgInlineStyle = null;
}

  function detectChatwootTheme() {
    // Chatwoot adiciona classe 'dark' no body quando está em dark mode
    return document.body.classList.contains('dark') ? 'dark' : 'light';
  }

  function applyThemeToPanel() {
    var panel = document.getElementById('cw-custom-panel');
    if (!panel) return;

    var theme = detectChatwootTheme();
    var isDark = theme === 'dark';

    console.log('[CW Theme] Aplicando tema:', theme);

    // Aplica tema no painel
    if (isDark) {
      panel.classList.add('dark-mode');
      panel.style.background = '#1f2937';

      // Atualiza loading overlay
      var loading = panel.querySelector('.loading-overlay');
      if (loading) {
        loading.style.background = '#1f2937';
        loading.style.color = '#e5e7eb';
      }

      var loadingText = panel.querySelector('.loading-text');
      if (loadingText) {
        loadingText.style.color = '#9ca3af';
      }

      // Atualiza error overlay
      var error = panel.querySelector('.error-overlay');
      if (error) {
        error.style.background = '#1f2937';
      }

      var errorTitle = panel.querySelector('.error-title');
      if (errorTitle) {
        errorTitle.style.color = '#f3f4f6';
      }

      var errorText = panel.querySelector('.error-text');
      if (errorText) {
        errorText.style.color = '#9ca3af';
      }
    } else {
      panel.classList.remove('dark-mode');
      panel.style.background = '#fff';

      var loading = panel.querySelector('.loading-overlay');
      if (loading) {
        loading.style.background = '#fff';
        loading.style.color = '#1f2d3d';
      }

      var loadingText = panel.querySelector('.loading-text');
      if (loadingText) {
        loadingText.style.color = '#6e84a3';
      }

      var error = panel.querySelector('.error-overlay');
      if (error) {
        error.style.background = '#fff';
      }

      var errorTitle = panel.querySelector('.error-title');
      if (errorTitle) {
        errorTitle.style.color = '#1f2d3d';
      }

      var errorText = panel.querySelector('.error-text');
      if (errorText) {
        errorText.style.color = '#6e84a3';
      }
    }

    // Envia mensagem para o iframe para aplicar tema também
    var iframe = document.getElementById('cw-iframe');
    if (iframe && iframe.contentWindow) {
      iframe.contentWindow.postMessage({
        action: 'setTheme',
        theme: theme
      }, '*');
    }
  }

  // Observa mudanças de tema no Chatwoot

    mutations.forEach(function(mutation) {
      if (mutation.type === 'attributes' && mutation.attributeName === 'class') {
        console.log('[CW Theme] Tema do Chatwoot mudou');
        applyThemeToPanel();
      }
    });
  });

  // Inicia observação do body para detectar mudanças de tema

    themeObserver.observe(document.body, {
      attributes: true,
      attributeFilter: ['class']
    });
    console.log('[CW Theme] Observer iniciado');
  }

  function getAuthFromCookie() {
    var cookies = document.cookie.split(';');
    for (var i = 0; i < cookies.length; i++) {
      var cookie = cookies[i].trim();
      if (cookie.indexOf('cw_d_session_info=') === 0) {
        try {
          var value = cookie.substring('cw_d_session_info='.length);
          var parsed = JSON.parse(decodeURIComponent(value));
          if (parsed['access-token'] && parsed['client'] && parsed['uid']) {
            return parsed;
          }
        } catch (e) {
          console.log('[CW Menu] Erro ao parsear cookie:', e);
        }
      }
    }
    return null;
  }

  function closeOnNavigation() {

  function setupNavigationDetection() {
    window.addEventListener('popstate', closeOnNavigation);
    var origPush = history.pushState;
    var origReplace = history.replaceState;
    history.pushState = function() { origPush.apply(this, arguments); closeOnNavigation(); };
    history.replaceState = function() { origReplace.apply(this, arguments); closeOnNavigation(); };
    document.addEventListener('click', function(e) {
      if (!panelOpen) return;
      var link = e.target.closest('a');
      if (link && !link.closest('#cw-custom-panel')) {
        setTimeout(closeOnNavigation, 100);
      }
    }, true);
  }

  async function checkIfSuperAdmin() {
    var authHeaders = getAuthFromCookie();
    if (!authHeaders) {
      console.log('[CW Menu] Sem authHeaders, não é SuperAdmin');
      return false;
    }

    try {
      // Usa URL relativa pois o script já está rodando no Chatwoot
      var res = await fetch('/api/v1/profile', {
        headers: {
          'access-token': authHeaders['access-token'],
          'token-type': authHeaders['token-type'] || 'Bearer',
          'client': authHeaders['client'],
          'expiry': authHeaders['expiry'] || '',
          'uid': authHeaders['uid']
        }
      });

      if (!res.ok) {
        console.log('[CW Menu] Erro na resposta do profile:', res.status);
        return false;
      }
      var data = await res.json();
      console.log('[CW Menu] Profile type:', data.type);
      return data.type === 'SuperAdmin';
    } catch (err) {
      console.log('[CW Menu] Erro ao verificar SuperAdmin:', err);
      return false;
    }
  }

  async function checkIfAccountAdmin() {
    var authHeaders = getAuthFromCookie();
    if (!authHeaders) {
      return false;
    }

    try {
      var res = await fetch('/api/v1/profile', {
        headers: {
          'access-token': authHeaders['access-token'],
          'token-type': authHeaders['token-type'] || 'Bearer',
          'client': authHeaders['client'],
          'expiry': authHeaders['expiry'] || '',
          'uid': authHeaders['uid']
        }
      });

      if (!res.ok) {
        return false;
      }
      var data = await res.json();
      console.log('[CW Menu] User role:', data.role);
      return data.role === 'administrator';
    } catch (err) {
      console.log('[CW Menu] Erro ao verificar role:', err);
      return false;
    }
  }

  async function checkEnabledFeatures() {
    var authHeaders = getAuthFromCookie();
    if (!authHeaders) {
      console.log('[CW Menu] Sem authHeaders, retornando features desabilitadas');
      return { kanbanEnabled: false, chatsInternosEnabled: false, conexoesEnabled: false, projectsEnabled: false, chatbotFlowsEnabled: false };
    }

    try {
      // Busca features habilitadas no backend KanbanCW
      var res = await fetch(KANBAN_URL + '/api/features', {
        headers: {
          'access-token': authHeaders['access-token'],
          'token-type': authHeaders['token-type'] || 'Bearer',
          'client': authHeaders['client'],
          'expiry': authHeaders['expiry'] || '',
          'uid': authHeaders['uid']
        }
      });

      if (!res.ok) {
        console.log('[CW Menu] Erro ao buscar features:', res.status);
        // Em caso de erro, habilita tudo (fail-open)
        return { kanbanEnabled: true, chatsInternosEnabled: true, conexoesEnabled: true };
      }

      var result = await res.json();
      var features = result.data || {};
      console.log('[CW Menu] Features habilitadas:', features);
      return {
        kanbanEnabled: features.kanbanEnabled !== false,
        chatsInternosEnabled: features.chatsInternosEnabled !== false,
        conexoesEnabled: features.conexoesEnabled !== false,
        projectsEnabled: features.projectsEnabled !== false,
        chatbotFlowsEnabled: features.chatbotFlowsEnabled !== false
      };
    } catch (err) {
      console.log('[CW Menu] Erro ao verificar features:', err);
      // Em caso de erro, habilita tudo (fail-open)
      return { kanbanEnabled: true, chatsInternosEnabled: true, conexoesEnabled: true, projectsEnabled: true, chatbotFlowsEnabled: true };
    }
  }

  async function checkUserPermissions() {
    try {
      var authHeaders = getAuthFromCookie();
      if (!authHeaders || !authHeaders['access-token']) {
        console.log('[CW Menu] Sem headers de auth, usando permissões padrão');
        return {
          kanbanAccess: true,
          conexoesAccess: true,
          chatsInternosAccess: true,
          projectsAccess: true,
          chatbotFlowsAccess: true,
          permissoesAccess: false
        };
      }

      var res = await fetch(KANBAN_URL + '/api/permissions/me', {
        headers: {
          'access-token': authHeaders['access-token'],
          'token-type': authHeaders['token-type'] || 'Bearer',
          'client': authHeaders['client'],
          'expiry': authHeaders['expiry'] || '',
          'uid': authHeaders['uid']
        }
      });

      if (!res.ok) {
        console.log('[CW Menu] Erro ao buscar permissões do usuário:', res.status);
        // Em caso de erro, habilita tudo (fail-open)
        return {
          kanbanAccess: true,
          conexoesAccess: true,
          chatsInternosAccess: true,
          projectsAccess: true,
          chatbotFlowsAccess: true,
          permissoesAccess: false
        };
      }

      var result = await res.json();
      var permissions = result.data || {};
      console.log('[CW Menu] Permissões do usuário:', permissions);
      return {
        kanbanAccess: permissions.kanbanAccess !== false,
        conexoesAccess: permissions.conexoesAccess !== false,
        chatsInternosAccess: permissions.chatsInternosAccess !== false,
        projectsAccess: permissions.projectsAccess !== false,
        chatbotFlowsAccess: permissions.chatbotFlowsAccess !== false,
        permissoesAccess: permissions.permissoesAccess === true
      };
    } catch (err) {
      console.log('[CW Menu] Erro ao verificar permissões:', err);
      // Em caso de erro, habilita tudo (fail-open)
      return {
        kanbanAccess: true,
        conexoesAccess: true,
        chatsInternosAccess: true,
        projectsAccess: true,
        chatbotFlowsAccess: true,
        permissoesAccess: false
      };
    }
  }

  window.wavoipCallCurrentContact = async function() {
    if (window._cwWavoipCalling) { console.log('[Wavoip] Chamada já em andamento, ignorando duplo clique'); return; }
    if (!currentConversationId) return;
    var authHeaders = getAuthHeaders();
    if (!authHeaders) return;
    window._cwWavoipCalling = true;
    try {
      var accountId = window.chatwootConfig?.accountId || 1;
      var res = await fetch(WIDGET_CONFIG.chatwootUrl + '/api/v1/accounts/' + accountId + '/conversations/' + currentConversationId, {
        headers: {
          'access-token': authHeaders['access-token'],
          'token-type': authHeaders['token-type'] || 'Bearer',
          'client': authHeaders['client'],
          'expiry': authHeaders['expiry'] || '',
          'uid': authHeaders['uid']
        }
      });
      if (!res.ok) { alert('Erro ao obter dados do contato.'); return; }
      var data = await res.json();
      var phone = data.meta?.sender?.phone_number;
      var contactName = data.meta?.sender?.name;
      if (!phone) { alert('Este contato não possui número de telefone cadastrado.'); return; }

      var iframe = document.getElementById('cw-iframe');
      if (!iframe) return;

      // Se o iframe não está carregado com o KanbanCW, abrir o painel e aguardar
      var isKanbanLoaded = iframe.src && iframe.src !== 'about:blank' && iframe.src.startsWith(KANBAN_URL);
      if (!isKanbanLoaded) {
        // Abre o painel para carregar o KanbanCW
        window.cwCustomOpen('');
        // Aguarda o iframe carregar
        await new Promise(function(resolve) {
          iframe.addEventListener('load', resolve, { once: true });
          setTimeout(resolve, 6000); // fallback
        });
        // Aguarda o WavoipContext inicializar
        await new Promise(function(r) { setTimeout(r, 2000); });
      }

      if (iframe.contentWindow) {
        iframe.contentWindow.postMessage({ type: 'WAVOIP_START_CALL', phone: phone, conversationId: currentConversationId, contactName: contactName }, '*');
      }
      // O widget flutuante será exibido quando o iframe confirmar WAVOIP_CALL_STARTED
      // Se a chamada falhar no SDK, o widget NÃO será exibido
    } catch (err) {
      console.error('[Wavoip] Erro ao iniciar chamada:', err);
      alert('Erro ao iniciar chamada. Verifique o console.');
    } finally {
      // Libera o lock após 3s (tempo para o SDK confirmar ou falhar)
      setTimeout(function() { window._cwWavoipCalling = false; }, 3000);
    }
  };

  async function loadWidgetModalData() {
    var body = document.getElementById('cw-kanban-modal-body');
    body.innerHTML = '<div class="loading"><div class="spinner"></div><span>Carregando...</span></div>';

    var authHeaders = getAuthHeaders();
    if (!authHeaders) {
      body.innerHTML = '<div class="error-msg">Token de autenticacao nao encontrado. Faca login novamente.</div>';
      return;
    }

    try {
      // Busca funis
      var funnelsRes = await fetch(WIDGET_CONFIG.apiUrl + '/funnels', {
        headers: Object.assign({ 'Content-Type': 'application/json' }, authHeaders)
      });

      // Detecta erro 401 especificamente (sessão expirada)
      if (funnelsRes.status === 401) {
        body.innerHTML = '<div style="padding: 40px 20px; text-align: center;"><div style="font-size: 48px; margin-bottom: 16px;">🔒</div><h3 style="font-size: 18px; font-weight: 600; color: #1f2d3d; margin: 0 0 12px;">Sessão Expirada</h3><p style="color: #6e84a3; margin: 0 0 24px; line-height: 1.6;">Sua sessão expirou. Por favor, recarregue a página para fazer login novamente.</p><button onclick="window.location.reload()" style="padding: 12px 24px; background: #1f93ff; color: white; border: none; border-radius: 8px; font-size: 14px; font-weight: 600; cursor: pointer; transition: all 0.2s;">🔄 Recarregar Página</button></div>';
        return;
      }

      if (!funnelsRes.ok) throw new Error('Erro ao carregar funis. Tente novamente.');

      var funnelsData = await funnelsRes.json();
      widgetModalState.funnels = funnelsData.data || [];

      // Busca stage atual se houver
      if (currentConversationId) {
        var stageRes = await fetch(WIDGET_CONFIG.apiUrl + '/kanban/conversation/' + currentConversationId + '/stage', {
          headers: Object.assign({ 'Content-Type': 'application/json' }, authHeaders)
        });
        if (stageRes.ok) {
          var stageData = await stageRes.json();
          widgetModalState.currentStage = stageData.data;
        }

        // Busca mensagens agendadas
        var scheduledRes = await fetch(WIDGET_CONFIG.apiUrl + '/conversations/' + currentConversationId + '/scheduled-messages', {
          headers: Object.assign({ 'Content-Type': 'application/json' }, authHeaders)
        });
        if (scheduledRes.ok) {
          var scheduledData = await scheduledRes.json();
          widgetModalState.scheduledMessages = scheduledData.data || [];
        }
      }

      renderWidgetModalContent();
    } catch (err) {
      body.innerHTML = '<div class="error-msg" style="padding: 20px;"><p style="margin: 0 0 12px;">' + err.message + '</p><button onclick="window.loadWidgetModalData()" style="padding: 8px 16px; background: #f0f0f0; border: 1px solid #ddd; border-radius: 6px; cursor: pointer; font-size: 13px;">Tentar Novamente</button></div>';
    }
  }

  function formatDateTime(dateString) {
    var date = new Date(dateString);
    return date.toLocaleString('pt-BR', {
      day: '2-digit',
      month: '2-digit',
      year: 'numeric',
      hour: '2-digit',
      minute: '2-digit'
    });
  }

  function getStatusLabel(status) {
    var labels = {
      'pending': 'Pendente',
      'sent': 'Enviada',
      'cancelled': 'Cancelada',
      'failed': 'Falhou'
    };
    return labels[status] || status;
  }

              }

              // Renderizar nodes em sequência
              nodes.forEach(function(node, index) {
                var nodeStatus = executedNodes[node.id];
                var isCurrent = node.id === currentNodeId;

                var nodeClass = 'flow-node';
                var nodeIcon = '○';

                if (nodeStatus === 'completed') {
                  nodeClass += ' flow-node-completed';
                  nodeIcon = '✓';
                } else if (nodeStatus === 'failed') {
                  nodeClass += ' flow-node-failed';
                  nodeIcon = '✗';
                } else if (nodeStatus === 'executing' || isCurrent) {
                  nodeClass += ' flow-node-current';
                  nodeIcon = '▶';
                } else if (nodeStatus === 'scheduled') {
                  nodeClass += ' flow-node-scheduled';
                  nodeIcon = '⏳';
                } else {
                  nodeClass += ' flow-node-pending';
                  nodeIcon = '○';
                }

                html += '<div class="' + nodeClass + '">';
                html += '<div class="flow-node-icon">' + nodeIcon + '</div>';
                html += '<div class="flow-node-label">' + getNodePreview(node) + '</div>';
                html += '</div>';

                // Adicionar connector se não for o último
                if (index < nodes.length - 1) {
                  html += '<div class="flow-connector"></div>';
                }
              });

              html += '</div>';
              html += '</div>';
            }
          } catch (e) {
            console.error('Erro ao renderizar diagrama do fluxo:', e);
          }
        }

        // Steps executados (lista detalhada)
        if (exec.steps && exec.steps.length > 0) {
          html += '<div class="report-section">';
          html += '<h4 class="report-subtitle">Histórico Detalhado</h4>';
          html += '<div class="steps-timeline">';

          exec.steps.forEach(function(step) {
            var stepStatusMap = {
              'scheduled': { label: 'Agendado', class: 'warning', icon: '📅' },
              'executing': { label: 'Executando', class: 'info', icon: '▶️' },
              'completed': { label: 'Concluído', class: 'success', icon: '✅' },
              'failed': { label: 'Falhou', class: 'error', icon: '❌' },
              'skipped': { label: 'Ignorado', class: 'default', icon: '⏭️' }
            };

            var stepStatus = stepStatusMap[step.status] || { label: step.status, class: 'default', icon: '❓' };

            html += '<div class="step-item step-' + step.status + '">';
            html += '<div class="step-icon">' + stepStatus.icon + '</div>';
            html += '<div class="step-content">';
            html += '<div class="step-type">' + step.nodeType + '</div>';

            if (step.scheduledFor && step.status === 'scheduled') {
              var schedDate = new Date(step.scheduledFor);
              html += '<div class="step-info">Agendado para: ' + schedDate.toLocaleString('pt-BR') + '</div>';
            }

            if (step.executedAt) {
              var execDate = new Date(step.executedAt);
              html += '<div class="step-info">Executado em: ' + execDate.toLocaleString('pt-BR') + '</div>';
            }

            if (step.error) {
              html += '<div class="step-error">Erro: ' + step.error + '</div>';
            }

            html += '<div class="step-status">' + stepStatus.label + '</div>';
            html += '</div>';
            html += '</div>';
          });

          html += '</div>';
          html += '</div>';
        } else {
          html += '<div class="empty-state">Nenhum step executado ainda</div>';
        }

        // Ações
        if (['pending', 'running', 'waiting', 'paused'].includes(exec.status)) {
          html += '<div class="report-actions">';
          if (exec.status === 'paused') {
            html += '<button class="btn-primary" onclick="window.resumeSequence(' + exec.id + ')">▶️ Retomar</button>';
          } else if (exec.status === 'waiting' || exec.status === 'running') {
            html += '<button class="btn-secondary" onclick="window.pauseSequence(' + exec.id + ')">⏸️ Pausar</button>';
          }
          html += '<button class="btn-danger" onclick="window.cancelSequence(' + exec.id + ')">❌ Cancelar</button>';
          html += '</div>';
        }
      }
    }

    body.innerHTML = html;
  }

    try {
      // Usa FormData para enviar arquivo
      var formData = new FormData();
      formData.append('message', message);
      formData.append('scheduledAt', new Date(scheduledAt).toISOString());

      if (attachmentEl && attachmentEl.files && attachmentEl.files[0]) {
        formData.append('attachment', attachmentEl.files[0]);
      }

        method: 'POST',
        headers: authHeaders, // Não inclui Content-Type para FormData

        throw new Error(errorData.error || 'Erro ao agendar mensagem');
      }

      var result = await res.json();
      widgetModalState.scheduledMessages.push(result.data);

      body.innerHTML = '<div class="success-msg">✓ Mensagem agendada com sucesso!</div>';
      setTimeout(function() {
        renderWidgetModalContent();

    if (!message && !hasAttachment) {
      alert('Por favor, digite uma mensagem ou anexe um arquivo');
      return;
    }

    if (!scheduledAt) {
      alert('Por favor, selecione data e hora');
      return;
    }

    var authHeaders = getAuthHeaders();
    if (!authHeaders || !currentConversationId) return;

    var body = document.getElementById('cw-kanban-modal-body');
    body.innerHTML = '<div class="loading"><div class="spinner"></div><span>Salvando...</span></div>';

    try {
      var formData = new FormData();
      formData.append('message', message);
      formData.append('scheduledAt', new Date(scheduledAt).toISOString());

      if (attachmentEl && attachmentEl.files && attachmentEl.files[0]) {
        formData.append('attachment', attachmentEl.files[0]);
      }

      if (widgetModalState.removeAttachmentFlag) {
        formData.append('removeAttachment', 'true');
      }

      var res = await fetch(WIDGET_CONFIG.apiUrl + '/conversations/' + currentConversationId + '/scheduled-messages/' + widgetModalState.editingMessage.id, {
        method: 'PATCH',
        headers: authHeaders,
        body: formData
      });

      if (!res.ok) {
        var errorData = await res.json();
        throw new Error(errorData.error || 'Erro ao atualizar mensagem');
      }

      var result = await res.json();

      // Atualiza na lista
      var index = widgetModalState.scheduledMessages.findIndex(function(m) { return m.id === result.data.id; });
      if (index !== -1) {
        widgetModalState.scheduledMessages[index] = result.data;
      }

      widgetModalState.editingMessage = null;
      body.innerHTML = '<div class="success-msg">✓ Mensagem atualizada com sucesso!</div>';
      setTimeout(function() {
        widgetModalState.view = 'scheduling';
        renderWidgetModalContent();
      }, 1500);
    } catch (err) {
      body.innerHTML = '<div class="error-msg">' + err.message + '</div><button class="btn-primary" style="margin-top:12px" onclick="window.editScheduledMessage(' + widgetModalState.editingMessage.id + ')">Voltar</button>';

  function checkConversationPageWidget() {
    var convId = getConversationIdFromUrl();

    if (convId) {
      currentConversationId = convId;
      var attempts = 0;
      var maxAttempts = 30;

      function tryAddButton() {
        console.log('[Kanban Widget] Tentativa ' + (attempts + 1) + ' de ' + maxAttempts);
        if (createHeaderButton()) {
          console.log('[Kanban Widget] Botão criado com sucesso!');
          checkCurrentStageWidget();
        } else {
          attempts++;
          if (attempts < maxAttempts) {
            setTimeout(tryAddButton, 500);
          } else {
            console.log('[Kanban Widget] Falhou após ' + maxAttempts + ' tentativas');
          }
        }
      }

      setTimeout(tryAddButton, 500);
    } else {
      currentConversationId = null;
      widgetModalState.currentStage = null;
      var btn = document.getElementById('cw-kanban-header-btn');
      if (btn) btn.remove();
      var callBtn = document.getElementById('cw-wavoip-call-btn');
      if (callBtn) callBtn.remove();
    }
  }

  function setupWidgetNavigationDetection() {
    var widgetLastUrl = location.href;

      if (location.href !== widgetLastUrl) {
        widgetLastUrl = location.href;
        checkConversationPageWidget();
      }
    }

    window.addEventListener('popstate', onWidgetUrlChange);

    setInterval(function() {
      if (location.href !== widgetLastUrl) {
        onWidgetUrlChange();
      }
    }, 500);
  }

(function () {
  'use strict';

  var SIDEBAR_TOOLS_SELECTOR = 'aside div.flex.gap-2.px-2';
  var WRAPPER_ID = 'cw-conversations-tools-wrapper';
  var SOURCE_ID = 'cw-sidebar-tools-source';
  var STYLE_ID = 'cw-tools-relocation-style';
  var HIDDEN_CLASS = 'cw-hide-tools';

  var syncQueued = false;

  function ensureStyles() {
    if (document.getElementById(STYLE_ID)) return;
    var s = document.createElement('style');
    s.id = STYLE_ID;
    s.textContent = '.cw-hide-tools{display:none !important;visibility:hidden !important;pointer-events:none !important;}';
    document.head.appendChild(s);
  }

  function queueSync() {
    if (syncQueued) return;
    syncQueued = true;
    requestAnimationFrame(function () {
      syncQueued = false;
      syncToolsPosition();
    });
  }

  function getSidebarToolsContainers() {
    var out = [];
    var candidates = document.querySelectorAll(SIDEBAR_TOOLS_SELECTOR);
    for (var i = 0; i < candidates.length; i++) {
      var c = candidates[i];
      if (c.querySelector('a[href*="/search"]') && c.querySelector('button')) out.push(c);
    }
    return out;
  }

  function findSidebarToolsContainer() {
    var containers = getSidebarToolsContainers();
    if (!containers.length) return document.getElementById(SOURCE_ID);

    // Prioriza o bloco do topo (próximo ao logo), evita pegar duplicatas no rodapé.
    var topCandidate = null;
    var topMin = Infinity;
    for (var i = 0; i < containers.length; i++) {
      var r = containers[i].getBoundingClientRect();
      if (r.top < topMin) {
        topMin = r.top;
        topCandidate = containers[i];
      }
    }

    if (topCandidate && !topCandidate.id) topCandidate.id = SOURCE_ID;
    return topCandidate || document.getElementById(SOURCE_ID);
  }

  function isConversationsRoute() {
    return /\/app\/accounts\/\d+\/(dashboard|inbox-view|inbox|conversations)(\/|$|\?)/.test(location.pathname + location.search);
  }

  function findConversationsTabsContainer() {
    // 1) estrutura semântica padrão
    var tablist = document.querySelector('[role="tablist"]');
    if (tablist) return tablist.parentElement || tablist;

    // 2) fallback por heading de Conversas
    var headings = document.querySelectorAll('h1,h2,h3,div,span');
    for (var h = 0; h < headings.length; h++) {
      var ht = (headings[h].textContent || '').trim().toLowerCase();
      if (ht !== 'conversas') continue;
      var container = headings[h].closest('section,main,article,div');
      for (var d = 0; container && d < 6; d++) {
        var ctext = (container.textContent || '').toLowerCase();
        if (ctext.indexOf('minhas') !== -1 || ctext.indexOf('não atribu') !== -1 || ctext.indexOf('nao atribu') !== -1 || ctext.indexOf('todos') !== -1) {
          return container;
        }
        container = container.parentElement;
      }
    }

    // 3) fallback por cabeçalho da coluna (tela Conversas)
    var convHeaders = document.querySelectorAll('h1,h2,h3');
    for (var ch = 0; ch < convHeaders.length; ch++) {
      var htxt = (convHeaders[ch].textContent || '').trim().toLowerCase();
      if (htxt !== 'conversas') continue;
      var col = convHeaders[ch].closest('section,article,main,div');
      if (col) return col;
    }

    // 4) fallback por texto das abas
    var seeds = document.querySelectorAll('a,button,span');
    for (var i = 0; i < seeds.length; i++) {
      var txt = (seeds[i].textContent || '').trim().toLowerCase();
      if (txt.indexOf('minhas') !== 0) continue;
      var scope = seeds[i].closest('div');
      for (var depth = 0; scope && depth < 6; depth++) {
        var t = (scope.textContent || '').toLowerCase();
        if (t.indexOf('minhas') !== -1) return scope;
        scope = scope.parentElement;
      }
    }
    return null;
  }

  function ensureWrapper(host) {
    var wrapper = document.getElementById(WRAPPER_ID);
    if (!wrapper) {
      wrapper = document.createElement('div');
      wrapper.id = WRAPPER_ID;
      wrapper.style.display = 'inline-flex';
      wrapper.style.alignItems = 'center';
      wrapper.style.gap = '8px';
      wrapper.style.marginLeft = 'auto';
      wrapper.style.flexWrap = 'nowrap';
    }
    if (wrapper.parentElement !== host) host.appendChild(wrapper);
    return wrapper;
  }

  function ensureProxyControls(wrapper, source) {
    if (!wrapper || !source) return;

    var sourceSearch = source.querySelector('a[href*="/search"]');
    var sourcePen = source.querySelector('button');
    if (!sourceSearch || !sourcePen) return;

    var proxySearch = wrapper.querySelector('a[data-cw-proxy="search"]');
    if (!proxySearch) {
      proxySearch = sourceSearch.cloneNode(true);
      proxySearch.setAttribute('data-cw-proxy', 'search');
      wrapper.appendChild(proxySearch);
    }
    proxySearch.href = sourceSearch.getAttribute('href') || '/app/accounts/1/search';

    var proxyPen = wrapper.querySelector('button[data-cw-proxy="pen"]');
    if (!proxyPen) {
      proxyPen = document.createElement('button');
      proxyPen.setAttribute('data-cw-proxy', 'pen');
      proxyPen.className = sourcePen.className;
      proxyPen.innerHTML = sourcePen.innerHTML;
      wrapper.appendChild(proxyPen);
      proxyPen.addEventListener('click', function (e) {
        e.preventDefault();
        e.stopPropagation();
        var src = document.getElementById(SOURCE_ID) ? document.getElementById(SOURCE_ID).querySelector('button') : null;
        if (src) src.click();
      });
    }
  }

  function showInConversations() {
    if (!isConversationsRoute()) return false;
    var source = findSidebarToolsContainer();
    var tabs = findConversationsTabsContainer();
    if (!source || !tabs) return false;

    var wrapper = ensureWrapper(tabs);
    ensureProxyControls(wrapper, source);

    // não remove do DOM: evita sumir ao trocar de rota
    source.style.visibility = 'hidden';
    source.style.pointerEvents = 'none';
    source.style.height = '0';
    source.style.overflow = 'hidden';
    source.style.margin = '0';
    source.style.padding = '0';
    return true;
  }

      all[i].style.pointerEvents = '';
      all[i].style.height = '';
      all[i].style.overflow = '';
      all[i].style.margin = '';

  }

    return /\/app\/accounts\/\d+\/search(\/|$|\?)/.test(location.pathname + location.search);
  }

    hideSidebarToolsOnly();
  }

  function scheduleSyncBurst() {
    setTimeout(queueSync, 80);
    setTimeout(queueSync, 250);
    setTimeout(queueSync, 700);

    setTimeout(queueSync, 3000);
  }

    ensureStyles();
    queueSync();

      attempts++;

    var tabsRow = findTabsRow();
    if (!tabsRow) return;
    var links = tabsRow.querySelectorAll('ul li a');
    links.forEach(function(a) {
      var txt = (a.textContent || '').trim();
      if (/^Minhas/i.test(txt)) {
        a.childNodes[0] && a.childNodes[0].nodeType === 3 ? a.childNodes[0].textContent = 'Atendendo ' : a.insertBefore(document.createTextNode('Atendendo '), a.firstChild);
      } else if (/^Não atribu/i.test(txt) || /^Nao atribu/i.test(txt)) {
        a.childNodes[0] && a.childNodes[0].nodeType === 3 ? a.childNodes[0].textContent = 'Aguardando ' : a.insertBefore(document.createTextNode('Aguardando '), a.firstChild);
      } else if (/^Todos/i.test(txt)) {
        a.childNodes[0] && a.childNodes[0].nodeType === 3 ? a.childNodes[0].textContent = 'Todas ' : a.insertBefore(document.createTextNode('Todas '), a.firstChild);
      }
    });
  }

  function mountHeaderActions() {
    var headerRow = findHeaderRow();
    if (!headerRow) return;

    var titleBlock = headerRow.querySelector('div.flex.items-center.justify-center.min-w-0');
    var actionsBlock = headerRow.querySelector('div.flex.items-center.gap-1');
    if (!titleBlock || !actionsBlock) return;

    var filterBtn = actionsBlock.querySelector('#toggleConversationFilterButton');
    var filterWrap = filterBtn ? filterBtn.closest('div') : null;
    var sortIcon = actionsBlock.querySelector('span.i-lucide-arrow-up-down');
    var sortBtn = sortIcon ? sortIcon.closest('button') : null;
    var sortWrap = sortBtn ? sortBtn.closest('div') : null;
    var layoutIcon = actionsBlock.querySelector('span.i-lucide-arrow-right-to-line');
    var layoutBtn = layoutIcon ? layoutIcon.closest('button') : null;

    var leftActions = headerRow.querySelector('[data-cw-left-actions]');
    if (!leftActions) {
      leftActions = document.createElement('div');
      leftActions.setAttribute('data-cw-left-actions', '1');
      titleBlock.appendChild(leftActions);
    }

    if (filterWrap && filterWrap.parentElement !== leftActions) leftActions.appendChild(filterWrap);
    if (sortWrap && sortWrap.parentElement !== leftActions) leftActions.appendChild(sortWrap);
    if (layoutBtn) layoutBtn.classList.add('cw-hidden-layout');

    var rightActions = document.getElementById(RIGHT_ACTIONS_ID);
    if (!rightActions) {
      rightActions = document.createElement('div');
      rightActions.id = RIGHT_ACTIONS_ID;
      headerRow.appendChild(rightActions);
    }

    if (!rightActions.querySelector('[data-cw-action="search"]')) {
      var searchBtn = document.createElement('button');
      searchBtn.className = 'cw-conv-icon-btn';
      searchBtn.setAttribute('data-cw-action', 'search');
      searchBtn.title = 'Pesquisar';
      searchBtn.innerHTML = '<span class="i-lucide-search"></span>';
      searchBtn.addEventListener('click', function () {
        var src = findSidebarToolsSource();
        var link = src ? src.querySelector('a[href*="/search"]') : null;
        if (link && link.href) window.location.href = link.href;
      });
      rightActions.appendChild(searchBtn);
    }

    if (!rightActions.querySelector('[data-cw-action="new-chat"]')) {
      var newBtn = document.createElement('button');
      newBtn.className = 'cw-conv-icon-btn';
      newBtn.setAttribute('data-cw-action', 'new-chat');
      newBtn.title = 'Nova mensagem';
      newBtn.innerHTML = '<span class="i-lucide-message-circle-plus"></span>';
      newBtn.addEventListener('click', function () {
        var src = findSidebarToolsSource();
        var btn = src ? src.querySelector('button') : null;
        if (btn) btn.click();
      });
      rightActions.appendChild(newBtn);
    }

    var legacy = document.getElementById('cw-conversations-tools-wrapper');
    if (legacy) legacy.style.display = 'none';
  }

  function applyConversationUI() {
    if (!isConversationsRoute()) return;
    ensureStyles();
    renameTabs();
    mountHeaderActions();
  }

  function schedule() {
    setTimeout(applyConversationUI, 20);
    setTimeout(applyConversationUI, 80);
    setTimeout(applyConversationUI, 300);
    setTimeout(applyConversationUI, 900);
  }

  var tabsObserver = null;
  function ensureTabsObserver() {
    if (tabsObserver) return;
    tabsObserver = new MutationObserver(function () {
      applyConversationUI();
    });
    tabsObserver.observe(document.body, { childList: true, subtree: true, characterData: true });
  }

  var push = history.pushState;
  history.pushState = function () { push.apply(this, arguments); schedule(); };
  var rep = history.replaceState;
  history.replaceState = function () { rep.apply(this, arguments); schedule(); };
  window.addEventListener('popstate', schedule);

  if (document.readyState === 'loading') document.addEventListener('DOMContentLoaded', function () { schedule(); ensureTabsObserver(); });
  else { schedule(); ensureTabsObserver(); }
})();
</script>
